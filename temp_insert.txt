def resolve_url(url: str, asset_prefix: str) -> str:
    if url.startswith(("http://", "https://", "tel:", "mailto:")):
        return url
    if url.startswith('#'):
        base = 'index.html' if asset_prefix else ''
        return f"{asset_prefix}{base}{url}" if asset_prefix else url
    if asset_prefix and not url.startswith('../'):
        return f"{asset_prefix}{url}"
    return url
